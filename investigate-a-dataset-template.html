def Rating_group (vote_average):
    if vote_average >=8:
        return 'High'
    elif vote_average >=5:
        return 'Medium'
    elif vote_average >=3:
        return 'Low'
    else:
        return 'Bad'

df['Rating_group'] = df['vote_average'].apply(Rating_group)
